<script>
//config
$float_speed=1500; //milliseconds
$float_easing="easeOutQuint";
$menu_fade_speed=500; //milliseconds
$closed_menu_opacity=0.75;
 
//cache vars
$fl_menu=$("#slide_nav");
$fl_menu_menu=$("#slide_nav .menu");
$fl_menu_label=$("#slide_nav .label");
$fl_menu_content=$("#slide_nav .content");

$(window).load(function() {
	menuPosition=$('#slide_nav').position().top;
	FloatMenu();
});
 
$(window).scroll(function () { 
	FloatMenu();
});
 
function FloatMenu(){
	var scrollAmount=$(document).scrollTop();
	var newPosition=scrollAmount;//menuPosition+scrollAmount
	if(newPosition > 200){
		$fl_menu.show();
	} else {
		$fl_menu.hide();
	}
	if($(window).height()<$fl_menu.height()+$fl_menu_menu.height()){
		$fl_menu.css("top",menuPosition);
	} else {
		$fl_menu.stop().animate({top: newPosition}, $float_speed, $float_easing);
	}
}

/* 鼠标移过显示不透明 */
$fl_menu_content.hover(function(){
	$(this).toggleClass('active')
})
</script>

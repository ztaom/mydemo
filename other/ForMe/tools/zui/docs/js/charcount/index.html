<script>
$(function(){
	var ajaxFormOpts = {clearForm:false, target:'#ajaxResponse', success: function(){$('#ajaxResponse').modal();}};
	$('form.ajaxForm').submit(function(){
		$("#loadingbar").show();
		if($(this).valid()){ $(this).ajaxSubmit(ajaxFormOpts);}	
		$('#ajaxResponse').hide();$("#loadingbar").hide();
		return false;
	});

});
</script>
